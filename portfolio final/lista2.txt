    <button onclick="jogar('papel')">Papel</button>
    <button onclick="jogar('tesoura')">Tesoura</button>
    <h3 id="resultado"></h3>
    <h4 id="mensagem"></h4>

    <script>
        function jogar(escolhaUsuario) {
            // Opções do computador
            const opcoes = ['pedra', 'papel', 'tesoura'];
            const escolhaComputador = opcoes[Math.floor(Math.random() * 3)];

            // Determina o vencedor
            let resultado = '';

            if (escolhaUsuario === escolhaComputador) {
                resultado = 'Empate!';
            } else if (
                (escolhaUsuario === 'pedra' && escolhaComputador === 'tesoura') ||
                (escolhaUsuario === 'papel' && escolhaComputador === 'pedra') ||
                (escolhaUsuario === 'tesoura' && escolhaComputador === 'papel')
            ) {
                resultado = 'Você ganhou!';
            } else {
                resultado = 'Você perdeu!';
            }

            // Exibe o resultado
            document.getElementById('resultado').innerText = `Você escolheu: ${escolhaUsuario}\nComputador escolheu: ${escolhaComputador}`;
            document.getElementById('mensagem').innerText = resultado;
        }
    </script>
</body>

            let numero = parseInt(document.getElementById('numero').value);
