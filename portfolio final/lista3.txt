        // Arrow functions para as operações matemáticas
        const somar = (a, b) => a + b;
        const subtrair = (a, b) => a - b;
        const multiplicar = (a, b) => a * b;
        const dividir = (a, b) => b !== 0 ? a / b : "Erro: Divisão por zero!";

        // Função para selecionar a operação e calcular o resultado
        function calcular(operacao) {
            let num1 = parseFloat(document.getElementById('num1').value);
            let num2 = parseFloat(document.getElementById('num2').value);
            let resultado;

            switch (operacao) {
                case '+':
                    resultado = somar(num1, num2);
                    break;
                case '-':
                    resultado = subtrair(num1, num2);
                    break;
                case '*':
                    resultado = multiplicar(num1, num2);
                    break;
                case '/':
                    resultado = dividir(num1, num2);
                    break;
                default:
                    resultado = "Operação inválida";
            }

            document.getElementById('resultado').innerText = `Resultado: ${resultado}`;
        }
    </script>

        const calcularCedulas = () => {
            let valor = parseInt(document.getElementById("valor").value);
            let cedulas = [200, 100, 50, 20, 10, 5, 1];
            let resultado = "";

            if (isNaN(valor) || valor <= 0) {
                document.getElementById("resultado").innerHTML = "Por favor, insira um valor válido!";
                return;
            }

            cedulas.forEach(cedula => {
                let qtd = Math.floor(valor / cedula);
                if (qtd > 0) {
                    resultado += `R$ ${cedula}: ${qtd} cédula(s) <br>`;
                    valor %= cedula;
                }
            });

            document.getElementById("resultado").innerHTML = resultado;
        };
    </script>

const calcularImpostoRenda = (salario) => {
    if (salario <= 1903.98) {
        return `Isento de Imposto de Renda.`;
    } else if (salario <= 2826.65) {
        return `Imposto: R$ ${(salario * 0.075 - 142.80).toFixed(2)}`;
    } else if (salario <= 3751.05) {
        return `Imposto: R$ ${(salario * 0.15 - 354.80).toFixed(2)}`;
    } else if (salario <= 4664.68) {
        return `Imposto: R$ ${(salario * 0.225 - 636.13).toFixed(2)}`;
    } else {
        return `Imposto: R$ ${(salario * 0.275 - 869.36).toFixed(2)}`;
    }
};

// Exemplo de uso:
let salario = 3000; // Altere o valor para testar
console.log(calcularImpostoRenda(salario));

const calcularINSS = (salario) => {
    let desconto;

    if (salario <= 1412.00) {
        desconto = salario * 0.075;
    } else if (salario <= 2666.68) {
        desconto = (1412.00 * 0.075) + ((salario - 1412.00) * 0.09);
    } else if (salario <= 4000.03) {
        desconto = (1412.00 * 0.075) + ((2666.68 - 1412.00) * 0.09) + ((salario - 2666.68) * 0.12);
    } else if (salario <= 7786.02) {
        desconto = (1412.00 * 0.075) + ((2666.68 - 1412.00) * 0.09) + ((4000.03 - 2666.68) * 0.12) + ((salario - 4000.03) * 0.14);
    } else {
        desconto = 908.86; // Teto máximo do INSS em 2024
    }

    return `Desconto do INSS: R$ ${desconto.toFixed(2)}`;
};

// Exemplo de uso:
let salario = 5000; // Altere o valor para testar
console.log(calcularINSS(salario));

const calcularJurosCompostos = (capital, taxa, tempo) => {
    let montante = capital * Math.pow((1 + taxa), tempo);
    return `Montante final: R$ ${montante.toFixed(2)}`;
};

// Exemplo de uso:
let capital = 1000; // Capital inicial R$
let taxa = 0.05;    // Taxa de 5% (0.05 em decimal)
let tempo = 12;     // Tempo em meses

console.log(calcularJurosCompostos(capital, taxa, tempo));

const calcularParcelaFinanciamento = (valor, taxa, parcelas) => {
    let i = taxa; // Taxa de juros decimal (ex: 5% = 0.05)
    let pmt = (valor * i) / (1 - Math.pow(1 + i, -parcelas));
    return `Valor da parcela: R$ ${pmt.toFixed(2)}`;
};

// Exemplo de uso:
let valorFinanciado = 50000; // Valor total do financiamento
let taxaJuros = 0.02;        // Taxa de 2% ao mês (0.02)
let numParcelas = 36;        // Número de parcelas

console.log(calcularParcelaFinanciamento(valorFinanciado, taxaJuros, numParcelas));
